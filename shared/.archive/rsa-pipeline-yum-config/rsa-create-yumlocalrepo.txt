#!/bin/sh
#################################################################################################
##
## This script is run only by root user
##
## Creates directories: /root/YumLocalRepo
## and sets default permissions for the created directory.
## 
##
## Fri May 31 2013 Vladimir Popov <vladimir.popov@duke.edu> 
## 
################################################################################################

echo
echo "=== Starting the $0 script =="
echo

if [ $USER != "root" ]; then
    echo "ERROR: this script must be executed under 'root' account."
    echo
    exit 1
fi

echo
echo "=== Creating and setting default permissions for directory /root/YumLocalRepo ==="
echo

# /root/YumLocalRepo
if [ -d "/root/YumLocalRepo" ]
then
    echo "Directory /root/YumLocalRepo exists.  Continue ..."
else
    echo "Directory /root/YumLocalRepo does not exist. Creating ... Continue ..."
    mkdir /root/YumLocalRepo
    chown root:root /root/YumLocalRepo
    chmod 755 /root/YumLocalRepo     
fi
echo
echo "=== Script finished ==="
echo
